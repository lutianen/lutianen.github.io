package bench1

import "testing"

func BenchmarkHorizontal(b *testing.B) {
	arrLen := 10000

	arr := make([][]int, arrLen, arrLen)

	for i := 0; i < arrLen; i++ {
		arrInternal := make([]int, arrLen)
		for j := 0; j < arrLen; j++ {
			arrInternal[j] = 0
		}
        arr[i] = arrInternal
	}

	for i := 0; i < b.N; i++ {
		for x := 0; x < len(arr); x++ {
			for y := 0; y < len(arr); y++ {
				arr[x][y] = 1
			}
		}
	}
}

func BenchmarkVertical(b *testing.B) {
	arrLen := 10000

	arr := make([][]int, arrLen, arrLen)

	for i := 0; i < arrLen; i++ {
		arrInternal := make([]int, arrLen)
		for j := 0; j < arrLen; j++ {
			arrInternal[j] = 0
		}
        arr[i] = arrInternal
	}

	for i := 0; i < b.N; i++ {
		for x := 0; x < len(arr); x++ {
			for y := 0; y < len(arr); y++ {
				arr[y][x] = 1
			}
		}
	}
}
